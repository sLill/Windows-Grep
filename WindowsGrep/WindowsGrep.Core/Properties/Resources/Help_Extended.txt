
-h	--help				: Show help. Use --help to view all commands
-r	--recursive			: Recursive search
-d	--directory=			: Set working directory (surround with ' or ")
-c	--context=			: Return n leading/trailing characters around the matched term
-b	--ignore-breaks			: Ignore line breaks
-i	--ignore-case			: Ignore case
-F	--fixed-string			: Interpret patterns as a fixed string instead of an expression
-t	--filetype-include=		: Include files of type n (, or ; delimited)
-T	--filetype-exclude=		: Exclude files of type n (, or ; delimited)
-p	--path-include=			: Include filepaths by expression (, or ; delimited)
-P	--path-exclude=			: Exclude filepaths by expression (, or ; delimited)
-k	--filenames			: Match file names
-o	--out-file=			: Redirect output to file
-v	--verbose			: Verbose output

--hash=					: Match by file hash (0=SHA256, 1=MD5)
--max-depth=				: Max search depth from the working directory when recursion (-r) is enabled
--show-hidden				: Show hidden files
--show-system				: Show system files
--filesize-min=				: Filter results < n(kb|mb|gb|tb)
--filesize-max=				: Filter results > n(kb|mb|gb|tb)

-RX	--replace=			: Replace text in matched files or filenames (surround with ' or ")
-DX	--delete			: Delete matched files
